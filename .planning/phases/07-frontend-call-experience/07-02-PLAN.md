```tsx
const OUTCOMES = [
  { value: "connected", label: "Connected", color: "bg-green-100 text-green-800" },
  { value: "voicemail", label: "Voicemail", color: "bg-yellow-100 text-yellow-800" },
  { value: "no_answer", label: "No Answer", color: "bg-red-100 text-red-800" },
  { value: "left_message", label: "Left Message", color: "bg-blue-100 text-blue-800" },
  { value: "wrong_number", label: "Wrong Number", color: "bg-gray-100 text-gray-800" },
  { value: "busy", label: "Busy", color: "bg-orange-100 text-orange-800" },
];

function OutcomeTagDropdown({ callId, currentOutcome }: { callId: string; currentOutcome: string | null }) {
  const { getToken } = useAuth();
  const queryClient = useQueryClient();
  const [open, setOpen] = React.useState(false);

  const handleSelect = async (value: string | null) => {
    const token = await getToken();
    if (!token) return;
    try {
      await updateCallOutcome(token, callId, value);
      queryClient.invalidateQueries({ queryKey: ["call-logs"] });
    } catch (e) {
      console.error("Failed to update outcome:", e);
    }
    setOpen(false);
  };

  const current = OUTCOMES.find((o) => o.value === currentOutcome);

  return (
    <div className="relative">
      <button
        onClick={() => setOpen(!open)}
        className={`inline-flex items-center gap-1 px-2 py-1 rounded text-xs font-medium ${
          current ? current.color : "bg-gray-50 text-gray-500 border border-gray-200"
        }`}
      >
        {current ? current.label : "Tag Outcome"}
        <ChevronDown className="w-3 h-3" />
      </button>
      {open && (
        <div className="absolute z-50 mt-1 w-40 bg-white rounded-lg shadow-lg border border-gray-200 py-1">
          {OUTCOMES.map((o) => (
            <button
              key={o.value}
              onClick={() => handleSelect(o.value)}
              className={`w-full text-left px-3 py-1.5 text-xs hover:bg-gray-50 ${
                o.value === currentOutcome ? "font-semibold" : ""
              }`}
            >
              <span className={`inline-block w-2 h-2 rounded-full mr-2 ${o.color.split(" ")[0]}`} />
              {o.label}
            </button>
          ))}
          {currentOutcome && (
            <button
              onClick={() => handleSelect(null)}
              className="w-full text-left px-3 py-1.5 text-xs text-gray-400 hover:bg-gray-50 border-t border-gray-100"
            >
              Clear
            </button>
          )}
        </div>
      )}
    </div>
  );
}
```

**3. Add `TranscriptSection` component** (if not already added by Phase 6):
```tsx
function TranscriptSection({ callId }: { callId: string }) {
  const { getToken } = useAuth();
  const queryClient = useQueryClient();
  const [expanded, setExpanded] = React.useState(false);

  const { data: transcript, isLoading } = useQuery({
    queryKey: ["call-transcript", callId],
    queryFn: async () => {
      const token = await getToken();
      return getCallTranscript(token!, callId);
    },
  });

  const handleConfirm = async (index: number) => {
    const token = await getToken();
    if (!token) return;
    await confirmTranscriptAction(token, callId, index);
    queryClient.invalidateQueries({ queryKey: ["call-transcript", callId] });
  };

  const handleDismiss = async (index: number) => {
    const token = await getToken();
    if (!token) return;
    await dismissTranscriptAction(token, callId, index);
    queryClient.invalidateQueries({ queryKey: ["call-transcript", callId] });
  };

  if (isLoading) return <div className="text-xs text-gray-400 py-2">Loading transcript...</div>;
  if (!transcript) return null;

  return (
    <div className="space-y-3">
      {/* AI Summary */}
      {transcript.ai_summary && (
        <div className="bg-blue-50 rounded-lg p-3">
          <div className="flex items-center gap-1.5 mb-1">
            <Bot className="w-3.5 h-3.5 text-blue-600" />
            <span className="text-xs font-semibold text-blue-800">AI Summary</span>
          </div>
          <p className="text-xs text-blue-900 leading-relaxed">{transcript.ai_summary}</p>
        </div>
      )}

      {/* AI Action Cards */}
      {transcript.ai_actions.filter((a: AIAction) => a.status === "pending").length > 0 && (
        <div className="space-y-2">
          <span className="text-xs font-semibold text-gray-600">Suggested Actions</span>
          {transcript.ai_actions.map((action: AIAction, idx: number) => (
            <AIActionCard
              key={idx}
              action={action}
              index={idx}
              onConfirm={handleConfirm}
              onDismiss={handleDismiss}
            />
          ))}
        </div>
      )}

      {/* Expandable Transcript */}
      <div>
        <button
          onClick={() => setExpanded(!expanded)}
          className="flex items-center gap-1 text-xs font-medium text-gray-600 hover:text-gray-900"
        >
          <FileText className="w-3.5 h-3.5" />
          Full Transcript ({transcript.word_count} words)
          {expanded ? <ChevronUp className="w-3 h-3" /> : <ChevronDown className="w-3 h-3" />}
        </button>
        {expanded && (
          <div className="mt-2 space-y-2 max-h-80 overflow-y-auto">
            {transcript.speaker_segments.map((seg: SpeakerSegment, idx: number) => (
              <div key={idx} className="text-xs">
                <span className={`font-semibold ${seg.speaker === "Agent" ? "text-blue-700" : "text-green-700"}`}>
                  {seg.speaker}:
                </span>
                <span className="text-gray-700 ml-1">{seg.text}</span>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
```

```tsx
function AIActionCard({ action, index, onConfirm, onDismiss }: {
  action: AIAction;
  index: number;
  onConfirm: (index: number) => void;
  onDismiss: (index: number) => void;
}) {
  const typeLabels: Record<string, string> = {
    create_task: "Create Task",
    update_buyer_profile: "Update Buyer Profile",
    update_deal_stage: "Update Deal Stage",
  };

  const typeColors: Record<string, string> = {
    create_task: "bg-purple-100 text-purple-800",
    update_buyer_profile: "bg-teal-100 text-teal-800",
    update_deal_stage: "bg-indigo-100 text-indigo-800",
  };

  return (
    <div className={`border rounded-lg p-3 ${
      action.status === "confirmed" ? "bg-green-50 border-green-200" :
      action.status === "dismissed" ? "bg-gray-50 border-gray-200 opacity-60" :
      "bg-white border-gray-200"
    }`}>
      <div className="flex items-center justify-between mb-1">
        <span className={`text-xs font-medium px-1.5 py-0.5 rounded ${typeColors[action.type] || "bg-gray-100 text-gray-800"}`}>
          {typeLabels[action.type] || action.type}
        </span>
        {action.status === "confirmed" && <CheckCircle className="w-4 h-4 text-green-600" />}
        {action.status === "dismissed" && <XCircle className="w-4 h-4 text-gray-400" />}

      <p className="text-xs text-gray-600 mb-2">
        {action.type === "create_task" && `Task: ${(action.params as Record<string, string>).body || "..."}`}

  );
}
```

```tsx
function CallDetailPanel({ call }: { call: CallLog }) {
  const formatDuration = (seconds: number) => {
    const m = Math.floor(seconds / 60);
    const s = seconds % 60;
    return `${m}:${s.toString().padStart(2, "0")}`;
  };

  return (
    <div className="space-y-4">
      {/* Header: direction, status, duration, timestamps */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <span className={`text-xs font-medium px-2 py-0.5 rounded ${
            call.direction === "inbound" ? "bg-green-100 text-green-800" : "bg-blue-100 text-blue-800"
          }`}>
            {call.direction === "inbound" ? "Inbound" : "Outbound"}
          </span>
          <span className="text-xs text-gray-500">{call.status}</span>
          {call.duration > 0 && (
            <span className="text-xs text-gray-500">{formatDuration(call.duration)}</span>
          )}
        </div>
        <OutcomeTagDropdown callId={call.id} currentOutcome={call.outcome} />
      </div>

      {/* AMD Badge */}
      {call.answered_by && call.answered_by.startsWith("machine") && (
        <div className="flex items-center gap-1.5 px-2 py-1 bg-yellow-50 border border-yellow-200 rounded text-xs text-yellow-800">
          <Bot className="w-3.5 h-3.5" />
          Voicemail Detected
        </div>
      )}

      {/* Recording Player */}
      {call.has_recording && (
        <div>
          <span className="text-xs font-semibold text-gray-600 block mb-1">Recording</span>
          <RecordingPlayer callId={call.id} />
        </div>
      )}

      {/* Transcript + AI Actions */}
      {(call.transcription_status === "completed" || call.has_recording) && (
        <div>
          <span className="text-xs font-semibold text-gray-600 block mb-1">Transcript & AI Insights</span>
          <TranscriptSection callId={call.id} />
        </div>
      )}
    </div>
  );
}
```
